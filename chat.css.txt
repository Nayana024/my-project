    /* ⭐ ONLY NEW FUNCTION YOU NEEDED ⭐ */
    function goBack(){
      window.location.href = "landing.html";
    }
 
    function filterVendors(){
      const q = document.getElementById('vendorSearch').value.trim().toLowerCase();
      const links = document.querySelectorAll('.vendor-link');
      const grid  = document.getElementById('vendorGrid');
 
      const oldMsg = document.getElementById('vendorEmptyMsg');
      if (oldMsg) oldMsg.remove();
 
      if (!q) {
        links.forEach(a => a.style.display = '');
        return;
      }
 
      let matches = 0;
      links.forEach(a => {
        const name = (a.getAttribute('data-name') || '').toLowerCase();
        const tag  = (a.getAttribute('data-tag')  || '').toLowerCase();
        const show = name.includes(q) || tag.includes(q);
        a.style.display = show ? '' : 'none';
        if (show) matches++;
      });
 
      if (matches === 0) {
        const msg = document.createElement('div');
        msg.id = 'vendorEmptyMsg';
        msg.textContent = 'Nothing found';
        msg.style.cssText = 'color:#fff;opacity:0.9;padding:20px;text-align:center;';
        grid.insertAdjacentElement('afterend', msg);
      }
    }
  </script>
